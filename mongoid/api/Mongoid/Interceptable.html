def run_callbacks(kind, *args, &block)
  cascadable_children(kind).each do |child|
    if child.run_callbacks(child_callback_type(kind, child), *args) == false
      return false
    end
  end
  if callback_executable?(kind)
    super(kind, *args, &block)
  else
    true
  end
end
