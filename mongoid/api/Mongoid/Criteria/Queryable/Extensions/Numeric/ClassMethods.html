def __numeric__(object)
  object.to_s =~ /(^[-+]?[0-9]+$)|(\.0+$)|(\.$)/ ? object.to_i : Float(object)
end

def evolve(object)
  __evolve__(object) do |obj|
    __numeric__(obj) rescue obj
  end
