def __expand_complex__
  replacement = {}
  each_pair do |key, value|
    replacement.merge!(key.__expr_part__(value.__expand_complex__))
  end
  replacement
end

def __sort_option__
  tap do |hash|
    hash.each_pair do |key, value|
      hash.store(key, value.to_direction)
    end
  end
end
