def __expr_part__(key, value, negating = false)
  if negating
    { key => { "$#{value.regexp? ? "not" : "ne"}" => value }}
  else
    { key => value }
  end
end

def evolve(object)
  __evolve__(object) do |obj|
    obj.regexp? ? obj : obj.to_s
  end
end
