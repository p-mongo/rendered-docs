def with_name(name)
  name_as_symbol = name.to_sym
  return clients[name_as_symbol] if clients[name_as_symbol]
  CREATE_LOCK.synchronize do
    clients[name_as_symbol] ||= Clients::Factory.create(name)
  end
end
