def create(attributes = nil, type = nil, &block)
  if attributes.is_a?(::Array)
    attributes.map { |attrs| create(attrs, type, &block) }
  else
    doc = build(attributes, type, &block)
    _base.persisted? ? doc.save : raise_unsaved(doc)
    doc
  end
end

def create!(attributes = nil, type = nil, &block)
  if attributes.is_a?(::Array)
    attributes.map { |attrs| create!(attrs, type, &block) }
  else
    doc = build(attributes, type, &block)
    _base.persisted? ? doc.save! : raise_unsaved(doc)
    doc
  end
end
