def delete(options = {})
  raise Errors::ReadonlyDocument.new(self.class) if readonly?
  prepare_delete do
    unless options[:persist] == false
      if embedded?
        delete_as_embedded(options)
      else
        delete_as_root
      end
    end
  end
end
