def able_to_set_updated_at?
  !frozen? && !timeless? && (new_record? || changed?)

def set_updated_at
  if able_to_set_updated_at?
    self.updated_at = Time.now.utc unless updated_at_changed?
  end

  clear_timeless_option
end
