def initialize(host, port, timeout, family, options = {})
  @host, @port, @timeout, @options = host, port, timeout, options
  @family = family
  @socket = ::Socket.new(family, SOCK_STREAM, 0)
  begin
    set_socket_options(@socket)
    connect!
  rescue
    @socket.close
    raise
  end
end
