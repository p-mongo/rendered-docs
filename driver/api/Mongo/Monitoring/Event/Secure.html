def redacted(command_name, document)
  if REDACTED_COMMANDS.include?(command_name.to_s) &&
    !%w(1 true yes).include?(ENV['MONGO_RUBY_DRIVER_UNREDACT_EVENTS']&.downcase)
  then
    BSON::Document.new
  else
    document
  end
end
